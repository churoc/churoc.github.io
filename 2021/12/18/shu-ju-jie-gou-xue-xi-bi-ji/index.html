#include<stdio.h>
#include<stdlib.h>
typedef char ElemType;

typedef struct BITNode
{
    ElemType data;
    struct BITNode *LChild, *RChild;
}BITNode, *BiTree;

//创建一个二叉树，约定前序遍历的方式输入
void CreateBiTree(BiTree *T)
{
    char c;
    scanf("%c", &c);

    if(' '==c)
    {
        *T=NULL;
    }
    else
    {
        *T=(BITNode *)malloc(sizeof(BITNode));
        (*T)->data=c;
        CreateBiTree(&(*T)->LChild);
        CreateBiTree(&(*T)->RChild);
    }
}

void visit(char c, int level)
{
    printf("%c 位于第%d层\n", c, level);
}

void PreOrderTraverse(BiTree T, int level)
{
    if(T)
    {
        visit(T->data, level);
        PreOrderTraverse(T->LChild, level+1);
        PreOrderTraverse(T->RChild, level+1);
    }
}

int main()
{
    int level=1;
    BiTree T=NULL;
    CreateBiTree(&T);
    PreOrderTraverse(T,level);
    return 0;
}

threaded_binary_tree

#include<stdio.h>
#include<stdlib.h>

typedef char ElemType;
//线索存储标识位
//Link(0):表示指向左右孩子的指针
//Thread(1)：表示指向前驱后继的线索
typedef enum 
{
    Link, Thread
}PointerTag;

typedef struct BiThrNode
{
    char data;
    struct BiThrNode *LChild, *RChild;
    PointerTag Ltag;
    PointerTag Rtag;
}BiThrNode, *BiThrTree;

//全局变量，始终指向刚刚访问过的结点
BiThrTree pre;

//创建一颗二叉树，约定用前序遍历输入数据
void CreateBiThrTree(BiThrTree *T)
{
    char c;
    scanf("%c", &c);
    if(' '== c)
    {
        *T=NULL;
    }
    else
    {
        *T = (BiThrNode *)malloc(sizeof(BiThrNode));
        (*T)->data= c;
        (*T)->Ltag= Link;
        (*T)->Rtag= Link;

        CreateBiThrTree(&(*T)->LChild);
        CreateBiThrTree(&(*T)->RChild);
    }
}

//中序遍历线索化
void InThreading(BiThrTree T)
{
    if(T)
    {
        InThreading(T->LChild);   //递归左孩子线索化

        //左空记录前驱，右空记录后继；
        if(!T->LChild)            //如果该结点没有左孩子，设置Ltag为Thread，并把LChild指向刚刚访问的结点
        {
            T->Ltag=Thread;
            T->LChild=pre;
        }

        if(!pre->RChild)
        {
            pre->Rtag=Thread;
            pre->RChild=T;
        }

        pre = T;                 

        InThreading(T->RChild);   //递归右孩子线索化
    }

}

//增加头结点；
void InOrderThreading(BiThrTree *p, BiThrTree T)
{
    *p= (BiThrNode *)malloc(sizeof(BiThrNode));
    
    (*p)->Rtag=Thread;
    (*p)->RChild=*p;

    (*p)->Ltag=Link;
    if(!T)
    {
        (*p)->LChild=*p;
    }
    else
    {
        (*p)->LChild=T;

        pre = *p;
        InThreading(T);
        
        pre->Rtag=Thread;     //收尾
        pre->RChild=*p;
        (*p)->RChild=pre;
    }
}

//中序遍历二叉树，非递归
void InOrderTraverse_Thr(BiThrTree T)
{
    BiThrTree p;
    p = T->LChild;
    while(p!=T)
    {
        while(p->Ltag==Link)
            p=p->LChild;
        printf("%c",p->data);
        while(p->Rtag==Thread && p->RChild!=T)
        {
            p=p->RChild;
            printf("%c",p->data);
        }
        p=p->RChild;
    }
}

int main()
{
    BiThrTree p, T = NULL;
    CreateBiThrTree(&T);
    InOrderThreading(&p, T);
    printf("输出结果是：");
    InOrderTraverse_Thr(p);
    return 0;
}

#include<stdio.h>

{
    int adjvex;
    EdgeType weight;
    struct EdgeNode *next;
}EdgeNode; 


typedef struct VertexNode                    //顶点表结点
{
    VertexType data;
    EdgeNode *firstedge;                     //边表头指针

typedef struct                               //图结构
{
    VertexNode VertexList[MAXVES];           //顶点表结点数组
    int numVertexes, numEdges;

void CreateAdjList(AdjList *G)
{
    int i, j, k;
    EdgeNode *e;
    printf("输入顶点数和边数\n");
    scanf("%d, %d", &G->numVertexes, &G->numEdges);

    for(i=0; i<G->numVertexes; i++)
    {
        scanf("%c",&G->VertexList[i].data);           //输入顶点信息
        G->VertexList[i].firstedge=NULL;         //边表置空
    }

    for(k=0; k<G->numEdges; k++)
    {
        printf("输入边的顶点序号\n");
        scanf("%d,%d", i, j);

        e=(EdgeNode*)malloc(sizeof(EdgeNode)); //生成边表结点
        if(!e)
            exit(0);
        //头插法
        e->adjvex=i;                           //边表结点的顶点数据输入
        e->next=G->VertexList[i].firstedge;       //边表结点的指向边表头指针
        G->VertexList[i].firstedge=e;             //修改边表头指针为e

        if(!e)
            exit(0);
        e->adjvex=j;
        e->next=G->VertexList[j].firstedge;
        G->VertexList[j].firstedge=e;

}

BFSTranverse_AdjMatric

#include<stdio.h>
#include<stdlib.h>
typedef char VertexType;
typedef int EdgeType;
typedef int ElemType;
#define MAXVEX 100
#define INFINITY 65535
#define FALSE 0
#define TRUE 1
#define MAX 10
int visit[MAX];

typedef struct QNode{
    ElemType data;
    struct QNode *next;
}QNode, *QueuePrt;//定义结点结构，结点指针

typedef struct{
    QueuePrt front, rear;
}LinkQueue;      //定义链式队列结构，结构包括队头指针和队尾指针

typedef struct
{
    VertexType vexs[MAXVEX];      //顶点表
    EdgeType arc[MAXVEX][MAXVEX]; //边表
    int numVertexes, numEdges;    //顶点数和边数
}MGraph;

void InitQueue(LinkQueue *q)
{
    q->front=q->rear=(QueuePrt)malloc(sizeof(QNode));
    if( !q->front )
        exit(0);
    q->front->next= NULL;
}

void InsertQueue(LinkQueue *q, ElemType e)
{
    QueuePrt p;
    p = (QueuePrt)malloc(sizeof(QNode));
    if(!p)
        exit(0);
    p->data = e;
    p->next = NULL;
    q->rear->next = p;
    q->rear = p;
}

void DeleteQueue(LinkQueue *q, ElemType *e)
{
    QueuePrt p;
    if(q->front == q->rear)
    {
        return;                           //?
    }
    p = q->front->next;
    *e = p->data;
    q->front->next = p->next;
    if( q->rear == p )
    {
        q->rear = q->front;
    }
    free(p);
    p = NULL;
}

int QueueEmpty(LinkQueue q)
{
    if(q.front == q.rear)
        return 1;
    else
        return 0;
}

void CreateMGraph(MGraph *G)
{
    int i, j, k;
    printf("输入顶点数和边数\n");
    scanf("%d,%d", &G->numVertexes, &G->numEdges);
    fflush(stdin);

    printf("输入顶点信息\n");
    for(i=0; i<G->numVertexes; i++)                 //读入顶点
    {
        scanf("%c", &G->vexs[i]);
    }

    fflush(stdin);
    for(i=0; i<G->numVertexes; i++)                 //初始化邻接矩阵
        for(j=0; j<G->numVertexes; j++)
        {
            G->arc[i][j]=INFINITY;
        }

    printf("请输入边的下标i，j和权值w\n");
    for(k=0; k<G->numEdges; k++)                    //读入边，建立邻接矩阵
    {
        scanf("%d, %d", &i, &j);
        G->arc[i][j]=1;
        G->arc[j][i]=G->arc[i][j];   
    }
}

void BFSTraverse(MGraph G)
{
    int i, j;
    LinkQueue Q;

    for(i=0; i<G.numVertexes; i++)
    {
        visit[i]=FALSE;
    }

    InitQueue(&Q);

    for(i=0; i<G.numVertexes; i++)
    {
        if(!visit[i])
        {
            printf("%c", G.vexs[i]);
            visit[i]=TRUE;
            InsertQueue(&Q, i);

            while(!QueueEmpty(Q))
            {
                DeleteQueue(&Q, &i);
                for(j=0; j<G.numVertexes; j++)
                {
                    if(G.arc[i][j]==1 && !visit[j])
                    {
                        printf("%c", G.vexs[j]);
                        visit[j]=TRUE;
                        InsertQueue(&Q, j);
                    }
                }
            }
        }
    }
}

int main()
{
    MGraph G;
    CreateMGraph(&G);
    BFSTraverse(G);
    return 0;
}

void DFS(AdjList G, int i)
{
    EdgeNode *p;
    visit[i]=TRUE;
    printf("%c", G.VertexList[i].data);

    p=G.VertexList[i].firstedge;
    while(p)
    {
        if(!visit[p->adjvex])
            DFS(G, p->adjvex);
        p=p->next;
    }
}

{
    int i;

        visit[i] = FALSE;
    for(i=0; i<G.numVertexes; i++)
        if(!visit[i])
            DFS(G, i);
}

#define EdgeMAX 15
typedef char VertexType;

    for(i=0; i<G->numVertexes; i++)                 //初始化邻接矩阵

        G->arc[i][j]=w;
        G->arc[j][i]=G->arc[i][j];    }

void TransEdge( EdgeList edge , MGraph G )
{
    int i , j  , k = 0  ;
    Edge t ;
    for( i = 0 ; i < G.numVertexes ; i++ )
        for(  j = i + 1 ; j < G.numVertexes ; j++ )
            if( G.arc[ i ][ j ] > 0 && G.arc[ i ][ j ] < INFINITY )
            {
                edge[ k ].begin = i ;
                edge[ k ].end = j ;
                edge[ k ].weight = G.arc[ i ][ j ] ;
                k++ ;
            }

    for( i = 0 ; i < G.numEdges - 1 ; i++ )
        for( j = 0 ; j < G.numEdges - i - 1 ; j++ )
            if( edge[ j ].weight > edge[ j + 1 ].weight )
            {
                t = edge[ j ] ;
                edge[ j ] = edge[ j+ 1 ] ;
                edge[ j+ 1 ] = t ;
            }
}

int Find(int *parent, int f)
{
    while(parent[f]>0)
        f=parent[f];
    return f;
}

void MiniSpanTree_Kruskal( MGraph G)  //每次选择最小权值的边，判断边的起始点是否同源，即选择此边后是否会形成闭环
{
    int i, m, n;
    EdgeList edge;
    int parent[MAXVEX];             //定义一个家谱

    TransEdge(edge, G);             //转换为边集数组，同时按权值进行排序

    for(i=0;i<G.numVertexes;i++)
        parent[i]=0;

    for(i=0;i<G.numEdges;i++)
    {
        n=Find(parent, edge[i].begin);   //找到begin点的祖宗
        m=Find(parent, edge[i].end);     //找到end点的祖宗

        if(n!=m)
        {
            parent[n]=m;
            printf("(%d,%d)-%d\n",edge[i].begin,edge[i].end, edge[i].weight);
        }
    }

    for(i=0;i<G.numVertexes;i++)
    {
        printf("%d",parent[i]);
    }
}

int main( void )
{
    MGraph G ;
    CreateMGraph( &G ) ;
    MiniSpanTree_Kruskal( G ) ;
    return 0 ;
}

    EdgeList edge;
    int parent[MAXVEX];             //定义一个家谱

    TransEdge(edge, G);             //转换为边集数组，同时按权值进行排序

    for(i=0;i<G.numVertexes;i++)
        parent[i]=0;

    for(i=0;i<G.numEdges;i++)
    {
        n=Find(parent, edge[i].begin);   //找到begin点的祖宗
        m=Find(parent, edge[i].end);     //找到end点的祖宗

        if(n!=m)
        {
            parent[n]=m;
            printf("(%d,%d)-%d\n",edge[i].begin,edge[i].end, edge[i].weight);
        }
    }

    for(i=0;i<G.numVertexes;i++)
    {
        printf("%d",parent[i]);
    }
}

int main( void )
{
    MGraph G ;
    CreateMGraph( &G ) ;
    MiniSpanTree_Kruskal( G ) ;
    return 0 ;
}

ShortestPath_Dijkstra

#include<stdio.h>
#define MAXVEX 9
#define INFINITY 65535
typedef char VertexType;
typedef int EdgeType;
typedef int Pathmatrix[MAXVEX];
typedef int ShortPathTable[MAXVEX];
typedef struct
{
    VertexType VexList[MAXVEX];      //顶点表
    EdgeType arc[MAXVEX][MAXVEX]; //边表
    int numVertexes, numEdges;    //顶点数和边数
}MGraph;

void CreateMGraph(MGraph *G)
{
    int i, j, k, w;
    printf("输入顶点数和边数\n");
    scanf("%d,%d", &G->numVertexes, &G->numEdges);
    fflush(stdin);

    printf("请输入顶点信息\n");
    for(i=0; i<G->numVertexes; i++)                 //读入顶点
    {
        scanf("%c", &G->VexList[i]);
    }
    fflush(stdin);

    for(i=0; i<G->numVertexes; i++)                 //初始化邻接矩阵
        for(j=0; j<G->numVertexes; j++)
        {
            G->arc[i][j]=INFINITY;
        }

    printf("请输入边的下标i，j和权值w\n");
    for(k=0; k<G->numEdges; k++)                    //读入边，建立邻接矩阵
    {
        scanf("%d, %d, %d", &i, &j, &w);
        G->arc[i][j]=w;
        G->arc[j][i]=G->arc[i][j];    }
}

void ShortestPath_Dijkstra(MGraph G, int v0, Pathmatrix *P, ShortPathTable *D)
{                                               //P记录上位点，D记录到v0的最短路径
    int v,w,min,k;
    int final[MAXVEX];                          //判断是否求得到某点的最短路径

    for(v=0; v<G.numVertexes; v++)
    {
        final[v]=0;                             //初始化为0，所有点最短路径都没有取得    
        (*D)[v]=G.arc[v0][v];
        (*P)[v]=0;
    }
    final[0]=1;                                 //v0点不需要求最短路径
    (*D)[v0]=0;

    for(v=1;v<G.numVertexes;v++)
    {
        min=INFINITY;
        for(w=0;w<G.numVertexes;w++)            //寻找离v0最近的点
        {
            if(!final[w] && (*D)[w]<min)
            {
                k=w;
                min=(*D)[w];
            }
        }
        printf("%d->%d\n",(*P)[k],k);
        final[k]=1;
        for(w=0;w<G.numVertexes;w++)            //更新到v0的最短路径，以及“上位点”
        {
            if(!final[w] && min+G.arc[k][w]<(*D)[w])
            {
                (*D)[w]=min+G.arc[k][w];
                (*P)[w]=k;
            }
        }
    }
}

int main()
{
    MGraph G;
    Pathmatrix P;
    ShortPathTable D;
    CreateMGraph(&G);
    ShortestPath_Dijkstra(G, 0, &P, &D);
    return 0;
}

ShortestPath_Floyd

#include<stdio.h>
#define MAXVEX 9
#define INFINITY 65535
typedef char VertexType;
typedef int EdgeType;
typedef int Pathmatrix[MAXVEX][MAXVEX];
typedef int ShortPathTable[MAXVEX][MAXVEX];
typedef struct
{
    VertexType VexList[MAXVEX];      //顶点表

void CreateMGraph(MGraph *G)
{
    int i, j, k, w;

    scanf("%d,%d", &G->numVertexes, &G->numEdges);

    }

    for(i=0; i<G->numVertexes; i++)                 //初始化邻接矩阵
        for(j=0; j<G->numVertexes; j++)
        {
            G->arc[i][j]=INFINITY;
        }

    printf("请输入边的下标i，j和权值w\n");
    for(k=0; k<G->numEdges; k++)                    //读入边，建立邻接矩阵

        G->arc[i][j]=w;
        G->arc[j][i]=G->arc[i][j];    }

void ShortPath_Floyd(MGraph G, Pathmatrix *P, ShortPathTable *D)
{
    int v, w, k;
    for(v=0;v<G.numVertexes;v++)
    {
        for(w=0;w<G.numVertexes;w++)
        {
            (*D)[v][w]=G.arc[v][w];
            (*P)[v][w]=w;
        }
    }

    for(k=0;k<G.numVertexes;k++)
    {
        for(v=0;v<G.numVertexes;v++)
        {
            for(w=0;w<G.numVertexes;w++)
            {
                if((*D)[v][w]>(*D)[v][k]+(*D)[k][w])
                {
                    (*D)[v][w]=(*D)[v][k]+(*D)[k][w];
                    (*P)[v][w]=(*P)[v][k];
                }
            }
        }
    }
}

int main()
{
    int v, w, k;
    MGraph G;
    Pathmatrix P;
    ShortPathTable D;
    CreateMGraph(&G);
    ShortPath_Floyd(G, &P, &D);
    
    printf("请输入起始点\n");
    scanf("%d%d",&v,&w);
    printf("\n%d -> %d 的最小路径为；%d\n", v, w, D[v][w]);
    k = P[v][w];
    printf("path: %d", v);//打印起点
    while(k != w)
    {
        printf("-> %d", k);//打印中间点
        k = P[k][w]; 
    }
    printf("-> %d\n", w);
    return 0;
}

Topological_Sorting_BFS

​ 该算法又称为kahn算法。
​ 如果顶点之间只是具有偏序关系，那么拓扑排序的结果肯定不唯一；如果顶点之间是全序关系，那么拓扑排序得到的序列唯一.

#include<stdio.h>
#include<stdlib.h>
#define OK 1
#define ERROR 0
#define Status int
typedef int VertexType;
typedef int EdgeType;
#define MAXVES 100

typedef struct EdgeNode                     //边表结点
{
    int adjvex;
    int weight;
    struct EdgeNode *next;
}EdgeNode; 

typedef struct VertexNode                    //顶点表结点
{
    VertexType data;
    int in;                                  //顶点入度
    EdgeNode *firstedge;                     //边表头指针
}VertexNode;

typedef struct                               //图结构
{
    VertexNode VertexList[MAXVES];           //顶点表结点数组
    int numVertexes, numEdges;
}AdjList;

int *etv, *ltv;
int *stack2;
int top2;

void CreateAdjList(AdjList *G)
{
    int i, j, k;
    EdgeNode *e;
    printf("输入顶点数和边数\n");
    scanf("%d, %d", &G->numVertexes, &G->numEdges);
    fflush(stdin);

    printf("输入顶点信息和他的入度\n");
    for(i=0; i<G->numVertexes; i++)
    {
        scanf("%d",&G->VertexList[i].data);      //输入顶点信息
        scanf("%d",&G->VertexList[i].in);
        G->VertexList[i].firstedge=NULL;         //边表置空
    }
    fflush(stdin);
    
    printf("输入边的顶点序号\n");
    for(k=0; k<G->numEdges; k++)
    {
        scanf("%d,%d", &i, &j);

        e=(EdgeNode*)malloc(sizeof(EdgeNode));   //生成边表结点
        if(!e)
            exit(0);
        //头插法
        e->adjvex=j;                             //边表结点的顶点数据输入
        e->next=G->VertexList[i].firstedge;      //边表结点的指向边表头指针
        G->VertexList[i].firstedge=e;            //修改边表头指针为e
    }
}  

Status TopologicalSort(AdjList G)
{
    EdgeNode *e;
    int i, k, gettop;
    int top =0;
    int count=0;
    int *stack;

    stack=(int *)malloc(G.numVertexes*sizeof(int));
    for(i=0; i<G.numVertexes; i++)
    {
        if(0==G.VertexList[i].in)
        {
            stack[++top]=i;
        }
    }

	while( 0 != top )
	{
		/* 出栈 */
		gettop = stack[top--];
        printf("%d -> ", G.VertexList[gettop].data);
		
		/* 记录拓扑排序输出顶点个数(用以判断是否存在环路) */
		count++;

		/* 取出一个入度为0的顶点后 以它为前驱的顶点入度-1 */
		for( e=G.VertexList[gettop].firstedge; e; e=e->next )
		{
			k = e->adjvex;
			/* 入度-1后若为0 则入栈 */
			if( !(--G.VertexList[k].in) )
			{
				stack[++top] = k;
			}
		}
	}

	/* 如果count小于顶点数 说明存在环 */
	if( count < G.numVertexes)
	{
		return ERROR;
	}
	else
	{
		return OK;
	}
}

int main()
{
	AdjList G;
	CreateAdjList(&G);
	TopologicalSort(G);
	return 0;
}

Status TopologicalSort(AdjList G)
{
    EdgeNode *e;
    int i, k, gettop;
    int top =0;
    int count=0;
    int *stack;
    stack=(int *)malloc(G.numVertexes*sizeof(int));
    //将入度为0的顶点入栈
    for(i=0; i<G.numVertexes; i++)
    {
        if(0==G.VertexList[i].in)
        {
            stack[++top]=i;
        }
    }

    //初始化
    top2 = 0; 
    etv=(int*)malloc(G.numVertexes*sizeof(int));
    for(i=0;i<G.numVertexes;i++)
    {
        etv[i]=0;
    }  
    stack2=(int*)malloc(G.numVertexes*sizeof(int));

	while( 0 != top )
	{
		/* 出栈 */
		gettop = stack[top--];
		/* 记录拓扑排序输出顶点个数(用以判断是否存在环路) */
		count++;
        stack2[++top2]=gettop;

		/* 取出一个入度为0的顶点后 以它为前驱的顶点入度-1 */
		for( e=G.VertexList[gettop].firstedge; e; e=e->next )
		{
			k = e->adjvex;
			/* 入度-1后若为0 则入栈 */
			if( !(--G.VertexList[k].in) )
			{
				stack[++top] = k;
			}
            if(etv[gettop]+e->weight > etv[k])
                etv[k]=etv[gettop]+e->weight;
		}
	}

	/* 如果count小于顶点数 说明存在环 */
	if( count < G.numVertexes)
	{
		return ERROR;
	}
	else
	{
		return OK;
	}
}

void CriticalPath(AdjList G)
{
    EdgeNode *e;
    int i, j, k, gettop;
    int ete, lte;
    TopologicalSort(G);
    ltv=(int*)malloc(G.numVertexes*sizeof(int));
    for(i=0;i<G.numVertexes;i++)
    {
        ltv[i]=etv[G.numVertexes-1];
    }
    while(0!=top2)
    {
        gettop=stack2[top2--];
        for(e=G.VertexList[gettop].firstedge;e;e=e->next)
        {
            k=e->adjvex;
            if(ltv[k]-e->weight<ltv[gettop])
                ltv[gettop]=ltv[k]-e->weight;
        }
    }

    for(j=0;j<G.numVertexes;j++)
    {
        for(e=G.VertexList[j].firstedge;e;e=e->next)   //遍历以vj为起点的所有活动
        {
            k=e->adjvex;
            ete=etv[j];
            lte=ltv[k]-e->weight;
            if(ete == lte)
                printf("<v%d,v%d>length:%d\n",G.VertexList[j].data,G.VertexList[k].data,e->weight);
        }
    }
}

Interpolation_Search

int Interpolation_Search(int *a, int n, int key)
{
    int low, high, mid;
    low=1;
    high=n;
    while (low<=high)
    {
        mid=low+(high-low)*(key-a[low])/(a[high]-a[low]);
        //当系数取1/2时为折半查找
        if(key==a[mid])
            return mid;
        else if(key>=mid)
        {
            low=mid+1;
        }
        else
        {
            high=mid-1; 
        }
    }
    return 0;
}

Fibonacci_Search

int F(int n)
{
    if(n==1||n==2)
        return 1;
    else    
        return F(n-1)+F(n-2);
}

int Fibonacci_Search(int *a, int n, int key)
{
    int low, high, mid, i, k;
    low=1;
    high=n;
    k=0;
    //补全k层fibonacci数
    while(n>F(k)-1)             //n从0开始计数，所以要减一
        k++;
    for(i=n;i<F(k)-1;i++)
        a[i]=a[n];

    while(low<=high)
    {
        mid=low+F(k-1)-1;        //mid取在0.618的位置
        if(key<a[mid])
        {
            high=mid-1;
            k=k-1;               //选择前面0.618分段继续查找
        }
        else if(key>a[mid])
        {
            low=mid+1;
            k=k-2;               //选择后面0.382分段继续查找
        }
        else
        {
            if(mid<=n)
                return mid;
            else
                return n;        //mid若在补全段，返回n
        }
    }
    return 0;
}

Bianry_Sort_Tree

#include<stdio.h>
#include<stdlib.h>
#define Status int
#define FALSE 0
#define TRUE 1
typedef int ElemType;
typedef struct BITNode
{
    ElemType data;
    struct BITNode *LChild, *RChild;
}BITNode, *BiTree;

//创建一个二叉树，约定前序遍历的方式输入
void CreateBiTree(BiTree *T)
{
    ElemType c;
    scanf("%d", &c);

    if(' '==c)
    {
        *T=NULL;
    }
    else
    {
        *T=(BITNode *)malloc(sizeof(BITNode));
        (*T)->data=c;
        CreateBiTree(&(*T)->LChild);
        CreateBiTree(&(*T)->RChild);
    }
}

//一些理解：
//树结构本身是一级结点指针，用BiTree找到根结点，用T->Child找到孩子结点，
//所以在函数传入值中，仅是调用的话，用一级结点指针，涉及创建，改动值的时候用二级结构指针，如*p。
//p指针存放查找路径上访问的最后一个结点
//f指向T的双亲。
Status SearchBST(BiTree T, int key, BiTree f, BiTree *p)
{
    if(!T)
    {
        *p=f;
        return FALSE;
    }
    else if(key==T->data)
    {
        *p=T;
        return TRUE;
    }
    else if (key<T->data)
        return SearchBST(T->LChild, key, T, p);
    else 
        return SearchBST(T->RChild, key, T, p);
}

Status Delete(BiTree *p)
{
    BiTree q, s;
    if((*p)->RChild==NULL)
    {
        q=*p;
        *p=(*p)->LChild;
        free(q);
    }
    else if((*p)->LChild==NULL)
    {
        q=*p;
        *p=(*p)->RChild;
        free(q);
    }
    else
    {
        q=*p;
        s=(*p)->LChild;
        while(s->RChild)
        {
            q=s;
            s=s->RChild;
        }
        (*p)->data=s->data;
        if(q!=*p)
            q->RChild=s->LChild;   
        //如果q不等于p，上面while起码遍历一次，删掉s，q的右孩子没了，把s原先左孩子接到q右侧。
        else
            q->LChild=s->LChild;   
        //如果q等于p，while没遍历，删掉s，q的左孩子没了，把s原先左孩子接到q左侧。
        free(s);
    }
    return TRUE;
}

Status InsertBST(BiTree *T, int key)
{
    BiTree p, s;
    if(!SearchBST(*T, key, NULL, &p))
    {
        s=(BiTree)malloc(sizeof(BITNode));
        s->data=key;
        s->LChild=s->RChild=NULL;
        if(!p)
            *T=s;
        else if(key<p->data)
            p->LChild=s;
        else    
            p->RChild=s;
        return TRUE;
    }
    else
        return FALSE;
}

Status DeleteBST(BiTree *T, int key)
{
    if(!*T)
        return FALSE;
    else
    {
        if(key==(*T)->data)
            return Delete(T);
        else if(key<(*T)->data)
            return DeleteBST(&(*T)->LChild, key);
        else
            return DeleteBST(&(*T)->RChild, key);  
    }
}
